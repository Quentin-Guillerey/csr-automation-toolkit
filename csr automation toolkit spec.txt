# csr_automation_toolkit.spec
# Build with: pyinstaller csr_automation_toolkit.spec
# (Rename this file back to .spec if it was pulled as _spec.txt.)
#
# Produces a single-file Windows .exe with no console window (GUI app),
# no Python install required on the teammate's machine.
#
# PATCH NOTE (2026-07):
# - 'openai' added to hiddenimports — it's imported lazily inside a
#   function, so make its inclusion explicit rather than relying on
#   PyInstaller's bytecode scan. The build machine must have openai,
#   gspread, and google-auth installed or the .exe ships permanently
#   missing those optional features.
# - 'oauth2client' removed (dependency dropped in favor of gspread's
#   native service-account auth).
# - upx=False: UPX-compressed unsigned executables are a classic
#   antivirus false-positive trigger; the size saving isn't worth
#   extra SmartScreen/AV friction during an internal pilot.

block_cipher = None

a = Analysis(
    ['csr_automation_toolkit.py'],
    pathex=[],
    binaries=[],
    datas=[],
    hiddenimports=[
        'gspread',
        'google.auth',
        'openai',
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[],
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=block_cipher,
    noarchive=False,
)

pyz = PYZ(a.pure, a.zipped_data, cipher=block_cipher)

exe = EXE(
    pyz,
    a.scripts,
    a.binaries,
    a.zipfiles,
    a.datas,
    [],
    name='CSR_Automation_Toolkit',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=False,
    upx_exclude=[],
    runtime_tmpdir=None,
    console=False,          # No black console window — GUI only
    disable_windowed_traceback=False,
    argv_emulation=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
    icon=None,              # Add icon='csr_icon.ico' here once you have one
)
